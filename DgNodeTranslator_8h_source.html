342 template<typename T>
343 MStatus DgNodeTranslator::setVec3Anim(MObject node, MObject attr, const UsdGeomXformOp op, double conversionFactor)
344 {
345  MPlug plug(node, attr);
346  MStatus status;
347  const char* const xformErrorCreate = "DgNodeTranslator:setVec3Anim error creating animation curve";
348 
349  MFnAnimCurve acFnSetX;
350  acFnSetX.create(plug.child(0), NULL, &status);
351  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
352 
353  MFnAnimCurve acFnSetY;
354  acFnSetY.create(plug.child(1), NULL, &status);
355  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
356 
357  MFnAnimCurve acFnSetZ;
358  acFnSetZ.create(plug.child(2), NULL, &status);
359  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
360 
361  std::vector<double> times;
362  op.GetTimeSamples(&times);
363 
364  const char* const xformErrorKey = "DgNodeTranslator:setVec3Anim error setting key on animation curve";
365 
366  T value(0);
367  for(auto const& timeValue: times)
368  {
369  const bool retValue = op.GetAs<T>(&value, timeValue);
370  if (!retValue) continue;
371 
372  MTime tm(timeValue, MTime::kFilm);
373 
374  switch (acFnSetX.animCurveType())
375  {
376  case MFnAnimCurve::kAnimCurveTL: // time->distance: translation
377  case MFnAnimCurve::kAnimCurveTA: // time->angle: rotation
378  case MFnAnimCurve::kAnimCurveTU: // time->double: scale
379  {
380  acFnSetX.addKey(tm, value[0] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
381  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
382  acFnSetY.addKey(tm, value[1] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
383  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
384  acFnSetZ.addKey(tm, value[2] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
385  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
386  break;
387  }
388  default:
389  {
390  break;
391  }
392  }
393  }
394 
395  return MS::kSuccess;
396 }
